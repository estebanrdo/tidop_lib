222

224 
225  if (i < posInBar) {
226  console.setBackgroundColor(mCompleteColor);
227  if (posInBar > ini && mCompleteColor == Console::Color::white)
228  console.setForegroundColor(Console::Color::black);
229  } else {
230  console.setBackgroundColor(mRemainigColor);
231  if (posInBar < ini + 4 && mRemainigColor == Console::Color::white)
232  console.setForegroundColor(Console::Color::black);
233  }
234 
235  int n;
236  if (i == ini) {
237  n = percent() / 100 % 10;
238  if (n > 0) std::cout << n;
239  else std::cout << " ";
240  } else if (i == ini + 1) {
241  n = percent() / 10 % 10;
242  if (n > 0 || percent() >= 10) std::cout << n;
243  else std::cout << " ";
244  } else if (i == ini + 2) {
245  n = percent() % 10;
246  std::cout << n;
247  } else if (i == ini + 3) {
248  std::cout << "%";
249  } else {
