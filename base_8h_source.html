867  class MatrixDerived, typename T, size_t Rows, size_t Cols>
868 auto MatrixBase<MatrixDerived<T, Rows, Cols>>::determinantnxn() const -> T
869 {
870  auto &derived = this->derived();
871 
872  T d = consts::one<T>;
873  size_t rows = derived.rows();
874  size_t cols = derived.cols();
875 
876  Matrix<T, DynamicData, DynamicData> matrix = derived;
877 
878  for (size_t i = 0; i < rows; ++i) {
879  T pivotElement = matrix(i, i);
880  size_t pivotRow = i;
881  for (size_t r = i + 1; r < rows; ++r) {
882  if (std::abs(matrix(r, i)) > std::abs(pivotElement)) {
883  pivotElement = matrix(r, i);
884  pivotRow = r;
885  }
886  }
887 
888  if (pivotElement == consts::zero<T>) {
889  d = consts::zero<T>;
890  break;
891  }
892 
893  if (pivotRow != i) {
894  matrix.swapRows(i, pivotRow);
895  d = -d;
896  }
897 
898  d *= pivotElement;
899 
900  for (size_t r = i + 1; r < rows; ++r) {
901  for (size_t c = i + 1; c < cols; ++c) {
902  matrix(r, c) -= matrix(r, i) * matrix(i, c) / pivotElement;
903  }
904  }
905  }
906 
907  return d;
