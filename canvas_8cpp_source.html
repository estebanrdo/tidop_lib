70 void CanvasCV::drawPoint(const Point<double> &point, const GraphicStyle &style)
71 {
72  const Symbol *style_symbol = style.symbol();
73  const Pen *style_pen = style.pen();
74  Color c = style_pen->color();
75  cv::Scalar color = colorToCvScalar(c);
76  Point<double> pt_offset(style_symbol->offsetX(), style_symbol->offsetY());
77  Point<int> pt = point + pt_offset;
78 
79  switch (style_symbol->name()) {
80  case Symbol::Name::cross:
81  cv::drawMarker(mCanvas, cv::Point(pt.x, pt.y), color, cv::MARKER_CROSS, 10, 1);
82  break;
83  case Symbol::Name::diagonal_cross:
84  cv::drawMarker(mCanvas, cv::Point(pt.x, pt.y), color, cv::MARKER_TILTED_CROSS, 10, 1);
85  break;
86  case Symbol::Name::circle:
87  cv::circle(mCanvas, cv::Point(pt.x, pt.y), 10, color, 1);
88  break;
89  case Symbol::Name::circle_filled:
90  cv::circle(mCanvas, cv::Point(pt.x, pt.y), 10, color, -1);
91  break;
92  case Symbol::Name::square:
93  cv::drawMarker(mCanvas, cv::Point(pt.x, pt.y), color, cv::MARKER_SQUARE, 10, 1);
94  break;
95  case Symbol::Name::square_filled:
96  cv::drawMarker(mCanvas, cv::Point(pt.x, pt.y), color, cv::MARKER_SQUARE, 10, -1);
97  break;
98  case Symbol::Name::triangle:
99  cv::drawMarker(mCanvas, cv::Point(pt.x, pt.y), color, cv::MARKER_TRIANGLE_UP, 10, 1);
100  break;
101  case Symbol::Name::triangle_filled:
102  cv::drawMarker(mCanvas, cv::Point(pt.x, pt.y), color, cv::MARKER_TRIANGLE_UP, 10, -1);
103  break;
104  case Symbol::Name::star:
105  cv::drawMarker(mCanvas, cv::Point(pt.x, pt.y), color, cv::MARKER_STAR, 10, 1);
106  break;
107  case Symbol::Name::star_filled:
108  cv::drawMarker(mCanvas, cv::Point(pt.x, pt.y), color, cv::MARKER_STAR, 10, -1);
109  break;
110  case Symbol::Name::vertical_bar:
111  break;
112  default:
113  cv::line(mCanvas, cv::Point(pt.x, pt.y), cv::Point(pt.x, pt.y), color, style_pen->width());
114  break;
115  }
116 
117  Label *style_label = style.label();
118  if (style_label && !style_label->text().empty()) {
119 
120  Color foregroundColor = style_label->foregroundColor();
121 
122 #ifdef HAVE_QT
123  Font font = style_label->font();
124  cv::QtFont qt_font = cv::fontQt(font.name(),font.size(),
125  colorToCvScalar(foregroundColor),
126  font.isBold() ? cv::QtFontWeights::QT_FONT_BOLD : cv::QtFontWeights::QT_FONT_NORMAL,
127  font.isItalic() ? cv::QtFontStyles::QT_STYLE_ITALIC : cv::QtFontStyles::QT_STYLE_NORMAL);
128  cv::addText(mCanvas, style_label->text(), cv::Point(pt.x, pt.y), qt_font);
129 #else
130  int fontFace = cv::FONT_HERSHEY_SIMPLEX;
131  double fontScale = 2;
132  int thickness = 3;
133  int baseline = 0;
134  //cv::Size textSize = cv::getTextSize(style_label->text(), fontFace,
135  // fontScale, thickness, &baseline);
136  baseline += thickness;
137 
138  cv::putText(mCanvas, style_label->text(), cv::Point(roundToInteger(point.x), roundToInteger(point.y)), fontFace, fontScale,
139  colorToCvScalar(foregroundColor), thickness, 8);
140 #endif
141 
142  }
