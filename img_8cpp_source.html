48 bool gdalValidExtensions(const std::string &extension)
49 {
50  std::vector<std::string> extensions {
51  ".bmp", // Microsoft Windows Device Independent Bitmap (.bmp)
52  ".png", // Portable Network Graphics (.png)
53  ".jpg", ".jpeg", ".jpe", ".jif", ".jfif", ".jfi", // JPEG JFIF (.jpg)
54  ".tif", ".tiff", // TIFF / BigTIFF / GeoTIFF (.tif)
55  ".gif", // Graphics Interchange Format (.gif)
56  ".gtx", // NOAA .gtx vertical datum shift
57  ".grd", // Arc/Info ASCII Grid
58  ".asc", // Arc/Info ASCII Grid
59  ".gsb", // NTv2 Datum Grid Shift
60  ".ecw", // ERDAS Compressed Wavelets (.ecw)
61  ".jp2", // JPEG2000 (.jp2, .j2k)
62  ".lan", // Erdas 7.x .LAN and .GIS
63  ".hdr", // ENVI .hdr Labelled Raster
64  ".img", // Erdas Imagine (.img)
65  ".blx", // Magellan BLX Topo (.blx, .xlb)
66  ".xlb", // Magellan BLX Topo (.blx, .xlb)
67  ".map", // OziExplorer .MAP
68  ".e00", // Arc/Info Export E00 GRID
69  ".hdr", // Vexcel MFF
70  ".img", // Erdas Imagine (.img)
71  ".wms"
72  };
73 
74  for (auto &gdal_extension : extensions) {
75  if (compareInsensitiveCase(extension, gdal_extension)) {
76  return true;
77  }
78  }
79 
80  return false;
81 }
