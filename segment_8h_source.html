255 template<typename Point_t>
256 Segment<Point_t>::Segment(const Point_t &pt,
257  double angle,
258  double length,
259  bool bCenter)
260  : Entity(Type::segment2d)
261 {
262 
263  double a = cos(angle);
264  double b = sin(angle);
265  double l1 = 0;
266  double l2 = length;
267 
268  if (bCenter) {
269  l1 = l2 = length / 2.;
270  }
271 
272  pt1.x = numberCast<scalar>(pt.x - l1 * -b);
273  pt1.y = numberCast<scalar>(pt.y - l1 * a);
274  pt2.x = numberCast<scalar>(pt.x + l2 * -b);
275  pt2.y = numberCast<scalar>(pt.y + l2 * a);
276 }

383 template<typename Point_t>
384 auto Segment<Point_t>::split(size_t n) const -> std::vector<Segment<Point_t>>
385 {
386  std::vector<Segment<Point_t>> segments;
387 
388  Point_t point1 = pt1;
389  Point_t point2;
390  for (size_t i = 1; i <= n; i++) {
391  point2 = pt1 * (1 - i / static_cast<double>(n)) + pt2 * i / static_cast<double>(n);
392  segments.emplace_back(point1, point2);
393  point1 = point2;
394  }
395 
396  return segments;
397 }
