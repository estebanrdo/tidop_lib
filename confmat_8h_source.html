180 template<typename T>
181 ConfusionMatrix<T>::ConfusionMatrix(const std::vector<std::pair<T, int>> &data)
182  : mData(data),
183  mPositives(0),
184  mNegatives(0)
185 {
186  std::sort(mData.begin(), mData.end(),
187  [](const std::pair<T, int> &data1,
188  const std::pair<T, int> &data2) {
189  return data1.first < data2.first; });
190 
191  for (const auto &pair : mData) {
192  if (pair.second == 1) {
193  mPositives++;
194  } else {
195  mNegatives++;
196  }
197  }
198 
199 }

349 template<typename T>
350 auto ConfusionMatrix<T>::compute(T threshold) const -> std::map<Classification, size_t>
351 {
352  size_t true_positives = 0;
353  size_t false_positives = 0;
354  size_t true_negatives = 0;
355  size_t false_negatives = 0;
356 
357 #if CPP_VERSION >= 17
358  for (const auto &[value, tag] : mData) {
359 #else
360  for (const auto &pair : mData) {
361  auto value = pair.first;
362  auto tag = pair.second;
363 #endif
364 
365  if (value < threshold) {
366  if (tag == 1)
367  false_negatives++;
368  else
369  true_negatives++;
370  } else {
371  if (tag == 0)
372  false_positives++;
373  else
374  true_positives++;
375  }
376  }
377 
378  std::map<Classification, size_t> confussionMatrix;
379  confussionMatrix[Classification::true_positives] = true_positives;
380  confussionMatrix[Classification::false_positives] = false_positives;
381  confussionMatrix[Classification::true_negatives] = true_negatives;
382  confussionMatrix[Classification::false_negatives] = false_negatives;
383 
384  return confussionMatrix;
385 }
