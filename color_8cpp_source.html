122 void rgbToCmyk(int red, int green, int blue, double *cyan, double *magenta, double *yellow, double *key)
123 {
124  adjustRangeRGBA(&red, &green, &blue);
125  double rgb[3] = {red / 255., green / 255., blue / 255.};
126  double max = *std::max_element(rgb, rgb + 3);
127  *key = 1. - max;
128  if (*key == 1.) {
129  *cyan = *magenta = *yellow = 0.;
130  } else {
131  *cyan = 1. - rgb[0] / max;
132  *magenta = 1. - rgb[1] / max;
133  *yellow = 1. - rgb[2] / max;
134  }
135 }

148 void rgbToHSL(int red, int green, int blue, double *hue, double *saturation, double *lightness)
149 {
150  adjustRangeRGBA(&red, &green, &blue);
151  double rgb[3] = {red / 255., green / 255., blue / 255.};
152 
153  double max = *std::max_element(rgb, rgb + 3);
154  double min = *std::min_element(rgb, rgb + 3);
155  double delta = max - min;
156 
157  if (delta > 0.) {
158  if (max == rgb[0])
159  *hue = 60. * (fmod(((rgb[1] - rgb[2]) / delta), 6.));
160  else if (max == rgb[1])
161  *hue = 60. * (((rgb[2] - rgb[0]) / delta) + 2.);
162  else
163  *hue = 60. * (((rgb[0] - rgb[1]) / delta) + 4.);
164  } else {
165  *hue = 0.;
166  }
167 
168  if (*hue < 0.) *hue += 360.;
169 
170  *lightness = (max + min) / 2;
171 
172  if (delta == 0)
173  *saturation = 0.;
174  else
175  *saturation = delta / (1 - std::abs(2 * *lightness - 1));
176 
177  *saturation *= 100;
178  *lightness *= 100;
179 }

182 void hslToRgb(double hue, double saturation, double lightness, int *red, int *green, int *blue)
183 {
184  adjustRangeHSL(&hue, &saturation, &lightness);
185 
186  lightness /= 100.;
187  saturation /= 100.;
188  double chroma = (1 - std::abs(2 * lightness - 1)) * saturation;
189  double h = hue / 60.;
190  double x = chroma * (1 - fabs(fmod(h, 2) - 1));
191 
192  std::array<double, 3> _rgb = {0., 0., 0.};
193 
194  if (h >= 0 && h < 1) {
195  _rgb[0] = chroma;
196  _rgb[1] = x;
197  } else if (h >= 1 && h < 2) {
198  _rgb[0] = x;
199  _rgb[1] = chroma;
200  } else if (h >= 2 && h < 3) {
201  _rgb[1] = chroma;
202  _rgb[2] = x;
203  } else if (h >= 3 && h < 4) {
204  _rgb[1] = x;
205  _rgb[2] = chroma;
206  } else if (h >= 4 && h < 5) {
207  _rgb[0] = x;
208  _rgb[2] = chroma;
209  } else {
210  _rgb[0] = chroma;
211  _rgb[2] = x;
212  }
213 
214  double m = lightness - chroma / 2;
215  *red = roundToInteger((_rgb[0] + m) * 255);
216  *green = roundToInteger((_rgb[1] + m) * 255);
217  *blue = roundToInteger((_rgb[2] + m) * 255);
218 }
