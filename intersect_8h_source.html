50 template <typename Point_t>
51 int intersectSegments(const Segment<Point_t> &ln1, const Segment<Point_t> &ln2, Point_t *pt)
52 {
53  int iret = 0;
54  Point_t vs1;
55  Point_t vs2;
56  vs1 = ln1.vector();
57  vs2 = ln2.vector();
58  // si el producto vectorial de los vectores que unen ambos segmentos es 0 son paralelas
59  double cross_product = crossProduct(vs1, vs2);
60  if (cross_product != 0.) {
61  Point_t v11_12 = ln2.pt1 - ln1.pt1;
62  double t = crossProduct(v11_12, vs2) / cross_product;
63  double u = crossProduct(v11_12, vs1) / cross_product;
64  if (t >= 0. && t <= 1 && u >= 0. && u <= 1) {
65  if (typeid(typename Point_t::value_type) == typeid(int)) {
66  pt->x = roundToInteger(ln1.pt1.x + t * vs1.x);
67  pt->y = roundToInteger(ln1.pt1.y + t * vs1.y);
68  } else {
69  pt->x += static_cast<typename Point_t::value_type>(ln1.pt1.x + t * vs1.x);
70  pt->y += static_cast<typename Point_t::value_type>(ln1.pt1.y + t * vs1.y);
71  }
72  iret = 1;
73  }
74  }
75 
76  return(iret);
77 }
