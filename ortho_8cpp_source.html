257 void Orthorectification::init()

268  mWindowDtmTerrainExtension.pt2.y = mAffineDtmImageToTerrain.ty + mAffineDtmImageToTerrain.scaleY() * mDtmReader->rows();
269 
270  mDifferentialRectification = std::make_unique<DifferentialRectification>(mCameraPose.rotationMatrix(),
271  mCameraPose.position(),
272  focal());
273 
274  bool exist_nodata = false;
275  double nodata_value = mDtmReader->noDataValue(&exist_nodata);
276  if (exist_nodata) mNoDataValue = nodata_value;
277 
278  PointD center_project = imageToTerrain(mRectImage.window().center());
279 
280  WindowD w(center_project /*mDifferentialRectification->cameraPosition()*/,
281  mAffineDtmImageToTerrain.scaleX(),
282  mAffineDtmImageToTerrain.scaleY());
283  cv::Mat image = mDtmReader->read(w);
284  mIniZ = image.at<float>(0, 0);
285 
286  PointF principal_point = this->principalPoint();
287  mAffineImageToPhotocoordinates = Affine<PointI>(-principal_point.x, principal_point.y, 1, -1, 0);
288 
289  mRectImage = Rect<int>(0, 0, mCamera.width(), mCamera.height());
290 
291  RectI rect_full(PointI(), mDtmReader->cols(), mDtmReader->rows());
292 
293  mFootprint.push_back(imageToTerrain(mRectImage.topLeft()));
294  mFootprint.push_back(imageToTerrain(mRectImage.topRight()));
295  mFootprint.push_back(imageToTerrain(mRectImage.bottomRight()));
296  mFootprint.push_back(imageToTerrain(mRectImage.bottomLeft()));
297 
298  WindowD window_terrain = mFootprint.window();
299 
300  PointI window_dtm_image_pt1 = terrainToDTM(window_terrain.pt1);
301  PointI window_dtm_image_pt2 = terrainToDTM(window_terrain.pt2);
302 
303  RectI rect(window_dtm_image_pt1, window_dtm_image_pt2);
304  rect.normalized();
305  mRectDtm = intersect(rect, rect_full);
306  mDtm = mDtmReader->read(mRectDtm);
307 }

309 void Orthorectification::initUndistortCamera()
310 {
311  std::shared_ptr<tl::Calibration> calibration = mCamera.calibration();
312 
313  float focal_x = 1.f;
314  float focal_y = 1.f;
315  {
316  for (auto param = calibration->parametersBegin(); param != calibration->parametersEnd(); param++) {
317  tl::Calibration::Parameters parameter = param->first;
318  float value = static_cast<float>(param->second);
319  switch (parameter) {
320  case tl::Calibration::Parameters::focal:
321  focal_x = value;
322  focal_y = value;
323  break;
324  case tl::Calibration::Parameters::focalx:
325  focal_x = value;
326  break;
327  case tl::Calibration::Parameters::focaly:
328  focal_y = value;
329  break;
330  default:
331  break;
332  }
333  }
334  }
335 
336  float ppx = static_cast<float>(calibration->parameter(tl::Calibration::Parameters::cx));
337  float ppy = static_cast<float>(calibration->parameter(tl::Calibration::Parameters::cy));

390  undistort_calibration->setParameter(tl::Calibration::Parameters::cy, optCameraMat.at<float>(1, 2));
