123 auto StereoImage::run(const cv::Mat& left,
124  const cv::Mat& right) -> cv::Mat
125 {
126  cv::Mat stereo;
127 
128  try {
129 
130  TL_ASSERT(left.size() == right.size(), "Different image size");
131  TL_ASSERT(left.type() == right.type(), "Different image type");
132 
133  int height = left.rows;
134  int width = left.cols;
135  int type = left.type();
136 
137  if (mMode == Mode::normal) {
138 
139  stereo.create(height, width, type);
140  // Se puede hacer con la matriz Normal
141  cv::Mat in[] = {left, right};
142  cv::Mat out[] = {stereo};
143  int from_to[] = {2, 2, left.channels() + 1, 1, left.channels(), 0};
144  cv::mixChannels(in, 2, out, 1, from_to, 3);
145 
146  } else if (mMode == Mode::side_by_side) {
147 
148  stereo.create(height, 2 * width, type);
149  cv::Size sz(width * 2, height);
150  cv::resize(stereo, stereo, sz);
151  stereo.adjustROI(0, 0, 0, -width);
152  right.copyTo(stereo);
153  stereo.adjustROI(0, 0, -width, width);
154  left.copyTo(stereo);
155  stereo.adjustROI(0, 0, width, 0);
156 
157  } else {
158  stereo.create(height, width, type);
159 
160  parallel_for(0, static_cast<size_t>(height), [&](size_t r) {
161  cv::Vec3b bgrPxL, bgrPxR;
162  int imode = static_cast<int>(mMode) - 1;
163  for (int c = 0; c < width; c++) {
164  bgrPxL = left.at<cv::Vec3b>(static_cast<int>(r), c);
165  bgrPxR = right.at<cv::Vec3b>(static_cast<int>(r), c);
166  cv::Vec3b &bgrSt = stereo.at<cv::Vec3b>(static_cast<int>(r), c);
167  bgrSt = AnaglyphMatrix[imode].imgRight * bgrPxR + AnaglyphMatrix[imode].imgLeft * bgrPxL;
168  }
169  });
170 
171  }
172 
173  } catch (...) {
174  std::throw_with_nested(std::runtime_error("StereoImage::run() failed"));
175  }
176 
177  return stereo;
178 }
