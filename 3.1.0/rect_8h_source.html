393 template<typename T>
394 auto intersect(const Rect<T> &rect1, const Rect<T> &rect2) -> Rect<T>
395 {
396  Rect<T> rect;
397 
398  rect.x = std::max(rect1.x, rect2.x);
399  rect.y = std::max(rect1.y, rect2.y);
400  Point<T> bottomRight1 = rect1.bottomRight();
401  Point<T> bottomRight2 = rect2.bottomRight();
402  rect.width = std::min(bottomRight1.x, bottomRight2.x) - rect.x;
403  rect.height = std::min(bottomRight1.y, bottomRight2.y) - rect.y;
404  if (rect.width < 0) rect.width = 0;
405  if (rect.height < 0) rect.height = 0;
406 
407  return rect;
