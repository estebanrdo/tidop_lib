105 template<typename Point_t>
106 double distPointToLine(const Point_t &pt, const Segment<Point_t> &ln)
107 {
108  Point_t v1 = vector2D(ln.pt1, pt);
109  Point_t v2 = ln.vector();
110  double b = dotProduct(v1, v2) / dotProduct(v2, v2);
111  Point_t pt2;
112  pt2.x = ln.pt1.x + b * v2.x;
113  pt2.y = ln.pt1.y + b * v2.y;
114  return distance(pt, pt2);
115 }

123 template<typename Point_t>

125 {

142 {
143  double max_dist = TL_DOUBLE_MAX;
144  double dist;
145  Polygon<Point_t> _p = polygon;
146  for (size_t i = 0; i < polygon.size(); i++) {
147  if (i == polygon.size() - 1) {
148  dist = distPointToSegment(pt, Segment<Point_t>(_p[i], _p[0]));

150  dist = distPointToSegment(pt, Segment<Point_t>(_p[i], _p[i + 1]));
151  }
152  if (dist < max_dist) max_dist = dist;
153  }

213 template<typename Point_t>
214 double distantePointToPlane(const Point_t &pt, const std::array<double, 4> &plane)
215 {
216  double num = plane[0] * pt.x + plane[1] * pt.y + plane[2] * pt.z + plane[3];
217  double normal = sqrt(plane[0] * plane[0] + plane[1] * plane[1] + plane[2] * plane[2]);
218  if (normal == 0.) throw std::runtime_error("3 puntos alineados");
219  return(num / normal);
220 }
