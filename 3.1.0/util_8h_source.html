54 template<typename Point_t> inline
55 Point_t findInscribedCircleSequential(const Polygon<Point_t> &polygon,
56  const Window<Point_t> bounds,
57  double nCells,
58  double mCells)
59 {
60  Point_t pia = bounds.center();
61  Point_t tmp{};
62 
63  // calculate the required increment for x and y
64  double increment_x = (bounds.pt2.x - bounds.pt1.x) / nCells;
65  double increment_y = (bounds.pt2.y - bounds.pt1.y) / mCells;
66 
67  // biggest known distance
68  double max_distance = 0.;
69 
70  double tmp_distance = TL_DOUBLE_MAX;
71  for (int i = 0; static_cast<double>(i) <= nCells; i++) {
72 
73  tmp.x = bounds.pt1.x + i * increment_x;
74 
75  for (int j = 0; static_cast<double>(j) <= mCells; j++) {
76 
77  tmp.y = bounds.pt1.y + j * increment_y;
78 
79  if (polygon.isInner(tmp)) {
80  tmp_distance = distPointToPolygon(tmp, polygon);
81  if (tmp_distance > max_distance) {
82  max_distance = tmp_distance;
83  pia.x = tmp.x;
84  pia.y = tmp.y;
85  }
86  }
87  }
88  }
89 
90  return pia;
91 }
