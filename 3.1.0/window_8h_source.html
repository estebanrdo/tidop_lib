131 template<typename Point_t> template<typename T>
132 Window<Point_t>::Window(const Point_t &pt,
133  T width, T height)
134  : Entity(Type::window)
135 {
136  auto half_width = width / consts::two<scalar>;
137  auto half_height = height / consts::two<scalar>;
138 
139 #if (CPP_VERSION >= 17)
140  if constexpr (std::is_integral<scalar>::value) {
141 #else
142  if (std::is_integral<scalar>::value) {
143 #endif
144 
145  int dx = static_cast<int>(width) % 2;
146  int dy = static_cast<int>(height) % 2;
147 
148  this->pt1 = Point_t(pt.x - half_width,
149  pt.y - half_height);
150  this->pt2 = Point_t(pt.x + half_width + dx,
151  pt.y + half_height + dy);
152  } else {
153  this->pt1 = Point_t(pt.x - half_width,
154  pt.y - half_height);
155  this->pt2 = Point_t(pt.x + half_width,
156  pt.y + half_height);
157  }
158 }
