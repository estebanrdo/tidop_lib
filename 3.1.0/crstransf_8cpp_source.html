225 auto rotationEnuToEcef(double longitude, double latitude) -> RotationMatrix<double>
226 {
227  RotationMatrix<double> rotation_enu;
228 
229  double longitude_rad = longitude * consts::deg_to_rad<double>;
230  double latitude_rad = latitude * consts::deg_to_rad<double>;
231 
232  double sin_longitude = sin(longitude_rad);
233  double cos_longitude = cos(longitude_rad);
234  double sin_latitude = sin(latitude_rad);
235  double cos_latitude = cos(latitude_rad);
236 
237  rotation_enu[0][0] = -sin_longitude;
238  rotation_enu[0][1] = cos_longitude;
239  rotation_enu[0][2] = 0;
240  rotation_enu[1][0] = -sin_latitude * cos_longitude;
241  rotation_enu[1][1] = -sin_latitude * sin_longitude;
242  rotation_enu[1][2] = cos_latitude;
243  rotation_enu[2][0] = cos_latitude * cos_longitude;
244  rotation_enu[2][1] = cos_latitude * sin_longitude;
245  rotation_enu[2][2] = sin_latitude;
246 
247  return rotation_enu;
248 }
