378 template<typename T, size_t Dim>
379 template<size_t rows, size_t cols>
380 auto RotationEstimator<T, Dim>::estimate(const Matrix<T, rows, cols> &src,
381  const Matrix<T, rows, cols> &dst) -> Rotation<T, Dim>
382 {
383  static_assert(dimensions == 2, "Rotation estimator only for 2D Scaling");
384 
385  Rotation<T, Dim> rotation;
386 
387  try {
388 
389  TL_ASSERT(src.cols() == dimensions, "Invalid matrix columns size");
390  TL_ASSERT(dst.cols() == dimensions, "Invalid matrix columns size");
391  TL_ASSERT(src.rows() == dst.rows(), "Different matrix sizes. Size src = {} and size dst = {}", src.rows(), dst.rows());
392 
393  size_t size = src.rows() * dimensions;
394 
395  Matrix<double> A(size, 2, 0);
396  Vector<double> B(size);
397 
398  for (size_t i = 0, r = 0; i < src.rows(); i++, r++) {
399 
400  A(r, 0) = src(i, 0);
401  A(r, 1) = -src(i, 1);
402  B[r] = dst(i, 0);
403 
404  r++;
405 
406  A(r, 0) = src(i, 1);
407  A(r, 1) = src(i, 0);
408  B[r] = dst(i, 1);
409 
410  }
411 
412  SingularValueDecomposition<Matrix<double>> svd(A);
413  Vector<double> C = svd.solve(B);
414 
415  rotation[0][0] = C[0];
416  rotation[0][1] = -C[1];
417  rotation[1][0] = C[1];
418  rotation[1][1] = C[0];
419 
420  } catch (...) {
421  TL_THROW_EXCEPTION_WITH_NESTED("");
422  }
423 
424  return rotation;
425 }

427 template<typename T, size_t Dim>
428 auto RotationEstimator<T, Dim>::estimate(const std::vector<Point<T>> &src,
429  const std::vector<Point<T>> &dst) -> Rotation<T, Dim>
430 {
431  TL_ASSERT(src.size() == dst.size(), "Size of origin and destination points different");
432 
433  Matrix<T> src_mat(src.size(), dimensions);
434  Matrix<T> dst_mat(dst.size(), dimensions);
435 
436  for (size_t r = 0; r < src_mat.rows(); r++) {
437  src_mat[r][0] = src[r].x;
438  src_mat[r][1] = src[r].y;
439 
440  dst_mat[r][0] = dst[r].x;
441  dst_mat[r][1] = dst[r].y;
442  }
443 
444  return RotationEstimator<T, dimensions>::estimate(src_mat, dst_mat);
445 }
