62  - (35 * e * e * e * e / 3072) * std::sin(6 * lat));
63  pt_utm.x = k0 * N * (A + (1 - T + C) * A * A * A / 6
64  + (5 - 18 * T + T * T + 72 * C - 58 * e * e) * A * A * A * A * A / 120) + 500000;
