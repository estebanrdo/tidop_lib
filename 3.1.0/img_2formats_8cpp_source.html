272 auto TiffOptions::options(bool all) const -> std::map<std::string, std::string>
273 {
274  std::map<std::string, std::string> options;
275 
276  if (all || bTFW.first != bTFW.second) {
277  options["TFW"] = bTFW.second ? "YES" : "NO";
278  }
279 
280  if (all || bRPB.first != bRPB.second) {
281  options["RPB"] = bRPB.second ? "YES" : "NO";
282  }
283 
284  if (all || bRPCTX.first != bRPCTX.second) {
285  options["RPCTX"] = bRPCTX.second ? "YES" : "NO";
286  }
287 
288  if (all || bTiled.first != bTiled.second) {
289  options["TILED"] = bTiled.second ? "YES" : "NO";
290  }
291 
292  if (all || mBlockXSize.first != mBlockXSize.second) {
293  options["BLOCKXSIZE"] = std::to_string(mBlockXSize.second);
294  }
295 
296  if (all || mBlockYSize.first != mBlockYSize.second) {
297  options["BLOCKYSIZE"] = std::to_string(mBlockYSize.second);
298  }
299 
300  if (all || mNBits.first != mNBits.second) {
301  options["NBITS"] = std::to_string(mNBits.second);
302  }
303 
304  if (all || mJpegQuality.first != mJpegQuality.second) {
305  options["JPEG_QUALITY"] = std::to_string(mJpegQuality.second);
306  }
307 
308  if (all || mZLevel.first != mZLevel.second) {
309  options["ZLEVEL"] = std::to_string(mZLevel.second);
310  }
311 
312  if (all || mBigTiff.first != mBigTiff.second) {
313  switch (mBigTiff.second) {
314  case BigTiff::yes:
315  options["BigTiff"] = "YES";
316  break;
317  case BigTiff::no:
318  options["BigTiff"] = "NO";
319  break;
320  case BigTiff::if_needed:
321  options["BigTiff"] = "IF_NEEDED";
322  break;
323  case BigTiff::if_safer:
324  options["BigTiff"] = "IF_SAFER";
325  break;
326  }
327  }
328 
329  if (all || mCompress.first != mCompress.second) {
330 
331  switch (mCompress.second) {
332  case Compress::jpeg:
333  options["COMPRESS"] = "JPEG";
334  break;
335  case Compress::lzw:
336  options["COMPRESS"] = "LZW";
337  break;
338  case Compress::packbits:
339  options["COMPRESS"] = "PACKBITS";
340  break;
341  case Compress::deflate:
342  options["COMPRESS"] = "DEFLATE";
343  break;
344  case Compress::ccittrle:
345  options["COMPRESS"] = "CCITTRLE";
346  break;
347  case Compress::ccittfax3:
348  options["COMPRESS"] = "CCITTFAX3";
349  break;
350  case Compress::ccittfax4:
351  options["COMPRESS"] = "CCITTFAX4";
352  break;
353  case Compress::lzma:
354  options["COMPRESS"] = "LZMA";
355  break;
356 #ifdef TL_HAVE_GDAL
357 #if GDAL_VERSION_NUM >= GDAL_COMPUTE_VERSION(2,3,0)
358  case Compress::zstd:
359  options["COMPRESS"] = "ZSTD";
360  break;
361 #endif
362 #if GDAL_VERSION_NUM >= GDAL_COMPUTE_VERSION(2,4,0)
363  case Compress::lerc:
364  options["COMPRESS"] = "LERC";
365  break;
366  case Compress::lerc_deflate:
367  options["COMPRESS"] = "LERC_DEFLATE";
368  break;
369  case Compress::lerc_zstd:
370  options["COMPRESS"] = "LERC_ZSTD";
371  break;
372 #endif
373 #endif // TL_HAVE_GDAL
374  case Compress::webp:
375  options["COMPRESS"] = "WEBP";
376  break;
377  case Compress::none:
378  options["COMPRESS"] = "NONE";
379  break;
380  }
381  }
382 
383  if (all || mPhotometric.first != mPhotometric.second) {
384 
385  switch (mPhotometric.second) {
386  case Photometric::minisblack:
387  options["PHOTOMETRIC"] = "MINISBLACK";
388  break;
389  case Photometric::miniswhite:
390  options["PHOTOMETRIC"] = "MINISWHITE";
391  break;
392  case Photometric::rgb:
393  options["PHOTOMETRIC"] = "RGB";
394  break;
395  case Photometric::cmyk:
396  options["PHOTOMETRIC"] = "CMYK";
397  break;
398  case Photometric::ycbcr:
399  options["PHOTOMETRIC"] = "YCBCR";
400  break;
401  case Photometric::cielab:
402  options["PHOTOMETRIC"] = "CIELAB";
403  break;
404  case Photometric::icclab:
405  options["PHOTOMETRIC"] = "ICCLAB";
406  break;
407  case Photometric::itulab:
408  options["PHOTOMETRIC"] = "ITULAB";
409  break;
410  }
411  }
412 
413  if (all || mAlpha.first != mAlpha.second) {
414 
415  switch (mAlpha.second) {
416  case Alpha::yes:
417  options["ALPHA"] = "YES";
418  break;
419  case Alpha::non_premultiplied:
420  options["ALPHA"] = "NON-PREMULTIPLIED";
421  break;
422  case Alpha::premultiplied:
423  options["ALPHA"] = "PREMULTIPLIED";
424  break;
425  case Alpha::unspecified:
426  options["ALPHA"] = "UNSPECIFIED";
427  break;
428  }
429  }
430 
431  if (all || mProfile.first != mProfile.second) {
432  switch (mProfile.second) {
433  case Profile::gdal_geotiff:
434  options["PROFILE"] = "GDALGeoTIFF";
435  break;
436  case Profile::geotiff:
437  options["PROFILE"] = "GeoTIFF";
438  break;
439  case Profile::baseline:
440  options["PROFILE"] = "BASELINE";
441  break;
442  }
443  }
444 
445  if (all || mPixelType.first != mPixelType.second) {
446  switch (mPixelType.second) {
447  case PixelType::def:
448  options["PIXELTYPE"] = "DEFAULT";
449  break;
450  case PixelType::signedbyte:
451  options["PIXELTYPE"] = "SIGNEDBYTE";
452  break;
453  }
454  }
455 
456  if (all || mGeotiffKeysFlavor.first != mGeotiffKeysFlavor.second) {
457  switch (mGeotiffKeysFlavor.second) {
458  case GeotiffKeysFlavor::standard:
459  options["GEOTIFF_KEYS_FLAVOR"] = "STANDARD";
460  break;
461  case GeotiffKeysFlavor::esri_pe:
462  options["GEOTIFF_KEYS_FLAVOR"] = "ESRI_PE";
463  break;
464  }
465  }
466 
467  if (all || mInternalMask.first != mInternalMask.second) {
468  options["GDAL_TIFF_INTERNAL_MASK"] = std::to_string(mBlockXSize.second);
469  }
470 
471  return options;
472 }
