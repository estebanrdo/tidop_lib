70 {
71  int norm = cv::NORM_L2;
72  Norm norm_type = BruteForceMatcherProperties::normType();
73  if (norm_type == Norm::l1) {
74  norm = cv::NORM_L1;
75  } else if (norm_type == Norm::l2) {
76  norm = cv::NORM_L2;
77  } else if (norm_type == Norm::hamming) {
78  norm = cv::NORM_HAMMING;

83  try {
