258 auto Path::list(const std::string &extension) -> std::list<Path>
259 {
260  std::list<Path> list;
261 
262  fs::directory_iterator it_end;
263 
264  for (fs::directory_iterator it(mPath->ref()); it != it_end; ++it) {
265 
266  if (!fs::is_regular_file(it->status())) continue;
267 
268  fs::path _path = it->path();
269 
270  std::string extension_found = it->path().extension().string();
271 
272  if (compareInsensitiveCase(extension_found, extension)) {
273 
274  list.emplace_back(it->path().filename().string());
275  }
276  }
277 
278  return list;

441 auto Path::tempDirectory() -> Path
442 {
443  std::string temp_path;
444 
445 #ifdef WIN32
446  temp_path = std::tmpnam(nullptr);
447 #else
448  temp_path = fs::temp_directory_path().string();
449  temp_path.append("/tlXXXXXX");
450  std::vector<char> c_path(temp_path.begin(), temp_path.end());
451  c_path.push_back('\0');
452  int file_descriptor = mkstemp(c_path.data());
453  if (file_descriptor != -1) {
454  temp_path.assign(c_path.begin(), c_path.end() - 1);
455  } else {
456  temp_path = "";
457  Message::warning("Failed to create temporary directory");
458  }
459 #endif
460 
461  return Path(temp_path);
462 }
