267 inline auto Ellipsoid::primeVertical(double lat) const -> double
268 {
269  double e = eccentricity();
270  double sin_lat = sin(lat * consts::deg_to_rad<double>);
271  return a / std::sqrt(1 - e*e * sin_lat*sin_lat);
