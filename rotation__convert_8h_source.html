315 template<typename T, int... P>
316 void RotationConverter<T, P...>::convert(const RotationMatrix<T> &rotationMatrix,
317  Quaternion<T> &quaternion)
318 {
319  const T four{4};
320 
321  T r22 = rotationMatrix[2][2];
322  if (r22 <= consts::zero<T>) {
323 
324  T r11_r00 = rotationMatrix[1][1] - rotationMatrix[0][0];
325  if (r11_r00 <= consts::zero<T>) {
326  quaternion.x = sqrt((consts::one<T> -r22 - r11_r00) / four);
327  T qx4 = quaternion.x * four;
328  quaternion.y = (rotationMatrix[0][1] + rotationMatrix[1][0]) / qx4;
329  quaternion.z = (rotationMatrix[0][2] + rotationMatrix[2][0]) / qx4;
330  quaternion.w = (rotationMatrix[2][1] - rotationMatrix[1][2]) / qx4;
331  } else {
332  quaternion.y = sqrt((consts::one<T> -r22 + r11_r00) / four);
333  T qy4 = quaternion.y * four;
334  quaternion.x = (rotationMatrix[0][1] + rotationMatrix[1][0]) / qy4;
335  quaternion.z = (rotationMatrix[1][2] + rotationMatrix[2][1]) / qy4;
336  quaternion.w = (rotationMatrix[0][2] - rotationMatrix[2][0]) / qy4;
337  }
338 
339  } else {
340 
341  T r11_r00 = rotationMatrix[1][1] + rotationMatrix[0][0];
342  if (r11_r00 <= consts::zero<T>) {
343  quaternion.z = sqrt((consts::one<T> +r22 - r11_r00) / four);
344  T qz4 = quaternion.z * four;
345  quaternion.x = (rotationMatrix[0][2] + rotationMatrix[2][0]) / qz4;
346  quaternion.y = (rotationMatrix[1][2] + rotationMatrix[2][1]) / qz4;
347  quaternion.w = (rotationMatrix[1][0] - rotationMatrix[0][1]) / qz4;
348  } else {
349  quaternion.w = sqrt((consts::one<T> +r22 + r11_r00) / four);
350  T qw4 = quaternion.w * four;
351  quaternion.x = (rotationMatrix[2][1] - rotationMatrix[1][2]) / qw4;
352  quaternion.y = (rotationMatrix[0][2] - rotationMatrix[2][0]) / qw4;
353  quaternion.z = (rotationMatrix[1][0] - rotationMatrix[0][1]) / qw4;
354  }
355 
356  }
357 }

359 template<typename T, int... P>
360 void RotationConverter<T, P...>::convert(const Quaternion<T> &quaternion,
361  RotationMatrix<T> &rotationMatrix)
362 {
363  T _2x = consts::two<T> *quaternion.x;
364  T _2y = consts::two<T> *quaternion.y;
365  T _2z = consts::two<T> *quaternion.z;
366  T _2xx = _2x * quaternion.x;
367  T _2xy = _2x * quaternion.y;
368  T _2xz = _2x * quaternion.z;
369  T _2xw = _2x * quaternion.w;
370  T _2yy = _2y * quaternion.y;
371  T _2yz = _2y * quaternion.z;
372  T _2yw = _2y * quaternion.w;
373  T _2zz = _2z * quaternion.z;
374  T _2zw = _2z * quaternion.w;
375 
376  rotationMatrix[0][0] = consts::one<T> -_2yy - _2zz;
377  rotationMatrix[0][1] = _2xy - _2zw;
378  rotationMatrix[0][2] = _2xz + _2yw;
379  rotationMatrix[1][0] = _2xy + _2zw;
380  rotationMatrix[1][1] = consts::one<T> -_2xx - _2zz;
381  rotationMatrix[1][2] = _2yz - _2xw;
382  rotationMatrix[2][0] = _2xz - _2yw;
383  rotationMatrix[2][1] = _2yz + _2xw;
384  rotationMatrix[2][2] = consts::one<T> -_2xx - _2yy;
385 }

387 template<typename T, int... P>
388 void RotationConverter<T, P...>::convert(const Quaternion<T> &quaternion,
389  AxisAngle<T> &axisAngle)
390 {
391  T n2 = std::sqrt(quaternion.x * quaternion.x + quaternion.y * quaternion.y + quaternion.z * quaternion.z);
392  if (n2 > consts::zero<T>) {
393  Vector<T, 3> axis;
394  axis[0] = quaternion.x / n2;
395  axis[1] = quaternion.y / n2;
396  axis[2] = quaternion.z / n2;
397  axisAngle.setAxis(axis);
398  T angle = consts::two<T> *std::acos(tl::clamp(quaternion.w, -consts::one<T>, consts::one<T>));
399  axisAngle.setAngle(angle);
400  } else {
401  axisAngle.setAxis({consts::one<T>, consts::zero<T>, consts::zero<T>});
402  axisAngle.setAngle(consts::zero<T>);
403  }
404 }

406 template<typename T, int... P>
407 void RotationConverter<T, P...>::convert(const AxisAngle<T> &axisAngle,
408  Quaternion<T> &quaternion)
409 {
410  T a_2 = axisAngle.angle() * consts::one_half<T>;
411  T sin_a_2 = std::sin(a_2);
412  quaternion.x = sin_a_2 * axisAngle.axis(0);
413  quaternion.y = sin_a_2 * axisAngle.axis(1);
414  quaternion.z = sin_a_2 * axisAngle.axis(2);
415  quaternion.w = std::cos(a_2);
416 }

427 template<typename T, int... P>
428 void RotationConverter<T, P...>::convert(const AxisAngle<T> &axisAngle,
429  RotationMatrix<T> &rotationMatrix)
430 {
431  Vector<T, 3> axis = axisAngle.axis();
432  T axis_0 = axis[0];
433  T axis_1 = axis[1];
434  T axis_2 = axis[2];
435 
436  T ca = std::cos(axisAngle.angle());
437  T sa = std::sin(axisAngle.angle());
438  T _1mca = consts::one<T> -ca;
439  T xy1mca = axis_0 * axis_1 * _1mca;
440  T xz1mca = axis_0 * axis_2 * _1mca;
441  T yz1mca = axis_1 * axis_2 * _1mca;
442  T xsa = axis_0 * sa;
443  T ysa = axis_1 * sa;
444  T zsa = axis_2 * sa;
445 
446  rotationMatrix[0][0] = axis_0 * axis_0 * _1mca + ca;
447  rotationMatrix[0][1] = xy1mca - zsa;
448  rotationMatrix[0][2] = xz1mca + ysa;
449  rotationMatrix[1][0] = xy1mca + zsa;
450  rotationMatrix[1][1] = axis_1 * axis_1 * _1mca + ca;
451  rotationMatrix[1][2] = yz1mca - xsa;
452  rotationMatrix[2][0] = xz1mca - ysa;
453  rotationMatrix[2][1] = yz1mca + xsa;
454  rotationMatrix[2][2] = axis_2 * axis_2 * _1mca + ca;
455 }

657 template<typename T, int... P>
658 void RotationConverter<T, P...>::convert(const EulerAngles<T, P...> &eulerAngles,
659  RotationMatrix<T> &rotationMatrix)
660 {
661  //typename EulerAngles<T>::Axes axes = eulerAngles.axes;
662 
663  T c1 = cos(eulerAngles.x);
664  T c2 = cos(eulerAngles.y);
665  T c3 = cos(eulerAngles.z);
666 
667  T s1 = sin(eulerAngles.x);
668  T s2 = sin(eulerAngles.y);
669  T s3 = sin(eulerAngles.z);
670 
671  // Tait-Bryan angles
672  if (eulerAngles.axes == Axes::xyz) {
673 
674  rotationMatrix[0][0] = c2 * c3;
675  rotationMatrix[0][1] = -c2 * s3;
676  rotationMatrix[0][2] = s2;
677  rotationMatrix[1][0] = c1 * s3 + c3 * s1 * s2;
678  rotationMatrix[1][1] = c1 * c3 - s1 * s2 * s3;
679  rotationMatrix[1][2] = -c2 * s1;
680  rotationMatrix[2][0] = s1 * s3 - c1 * c3 * s2;
681  rotationMatrix[2][1] = c3 * s1 + c1 * s2 * s3;
682  rotationMatrix[2][2] = c1 * c2;
683 
684  } else if (eulerAngles.axes == Axes::yxz) {
685 
686  rotationMatrix[0][0] = c1 * c3 + s1 * s2 * s3;
687  rotationMatrix[0][1] = c3 * s1 * s2 - c1 * s3;
688  rotationMatrix[0][2] = c2 * s1;
689  rotationMatrix[1][0] = c2 * s3;
690  rotationMatrix[1][1] = c2 * c3;
691  rotationMatrix[1][2] = -s2;
692  rotationMatrix[2][0] = c1 * s2 * s3 - c3 * s1;
693  rotationMatrix[2][1] = c1 * c3 * s2 + s1 * s3;
694  rotationMatrix[2][2] = c1 * c2;
695 
696  } else if (eulerAngles.axes == Axes::zxy) {
697 
698  rotationMatrix[0][0] = c1 * c3 - s1 * s2 * s3;
699  rotationMatrix[0][1] = -s1 * c2;
700  rotationMatrix[0][2] = c1 * s3 + s1 * s2 * c3;
701  rotationMatrix[1][0] = s1 * c3 + c1 * s2 * s3;
702  rotationMatrix[1][1] = c1 * c2;
703  rotationMatrix[1][2] = s1 * s3 - c1 * s2 * c3;
704  rotationMatrix[2][0] = -c2 * s3;
705  rotationMatrix[2][1] = s2;
706  rotationMatrix[2][2] = c2 * c3;
707 
708  } else if (eulerAngles.axes == Axes::zyx) {
709 
710  rotationMatrix[0][0] = c1 * c2;
711  rotationMatrix[0][1] = c1 * s2 * s3 - c3 * s1;
712  rotationMatrix[0][2] = s1 * s3 + c1 * c3 * s2;
713  rotationMatrix[1][0] = c2 * s1;
714  rotationMatrix[1][1] = c1 * c3 + s1 * s2 * s3;
715  rotationMatrix[1][2] = c3 * s1 * s2 - c1 * s3;
716  rotationMatrix[2][0] = -s2;
717  rotationMatrix[2][1] = c2 * s3;
718  rotationMatrix[2][2] = c2 * c3;
719 
720  } else if (eulerAngles.axes == Axes::yzx) {
721 
722  rotationMatrix[0][0] = c1 * c2;
723  rotationMatrix[0][1] = s1 * s3 - c1 * s2 * c3;
724  rotationMatrix[0][2] = c1 * s2 * s3 + s1 * c3;
725  rotationMatrix[1][0] = s2;
726  rotationMatrix[1][1] = c2 * c3;
727  rotationMatrix[1][2] = -c2 * s3;
728  rotationMatrix[2][0] = -s1 * c2;
729  rotationMatrix[2][1] = c1 * s3 + s1 * s2 * c3;
730  rotationMatrix[2][2] = c1 * c3 - s1 * s2 * s3;
731 
732  } else if (eulerAngles.axes == Axes::xzy) {
733 
734  rotationMatrix[0][0] = c2 * c3;
735  rotationMatrix[0][1] = -s2;
736  rotationMatrix[0][2] = c2 * s3;
737  rotationMatrix[1][0] = c1 * s2 * c3 + s1 * s3;
738  rotationMatrix[1][1] = c1 * c2;
739  rotationMatrix[1][2] = c1 * s2 * s3 - s1 * c3;
740  rotationMatrix[2][0] = s1 * s2 * c3 - c1 * s3;
741  rotationMatrix[2][1] = s1 * c2;
742  rotationMatrix[2][2] = s1 * s2 * s3 + c1 * c3;
743 
744  }
745  // Euler angles
746  else if (eulerAngles.axes == Axes::xyx) {
747 
748  rotationMatrix[0][0] = c2;
749  rotationMatrix[0][1] = s2 * s3;
750  rotationMatrix[0][2] = s2 * c3;
751  rotationMatrix[1][0] = s1 * s2;
752  rotationMatrix[1][1] = c1 * c3 - s1 * c2 * s3;
753  rotationMatrix[1][2] = -c1 * s3 - s1 * c2 * c3;
754  rotationMatrix[2][0] = -c1 * s2;
755  rotationMatrix[2][1] = s1 * c3 + c1 * c2 * s3;
756  rotationMatrix[2][2] = c1 * c2 * c3 - s1 * s3;
757 
758  } else if (eulerAngles.axes == Axes::xzx) {
759 
760  rotationMatrix[0][0] = c2;
761  rotationMatrix[0][1] = -c3 * s2;
762  rotationMatrix[0][2] = s2 * s3;
763  rotationMatrix[1][0] = c1 * s2;
764  rotationMatrix[1][1] = c1 * c2 * c3 - s1 * s3;
765  rotationMatrix[1][2] = -c3 * s1 - c1 * c2 * s3;
766  rotationMatrix[2][0] = s1 * s2;
767  rotationMatrix[2][1] = c1 * s3 + c2 * c3 * s1;
768  rotationMatrix[2][2] = c1 * c3 - c2 * s1 * s3;
769 
770  } else if (eulerAngles.axes == Axes::yxy) {
771 
772  rotationMatrix[0][0] = c1 * c3 - c2 * s1 * s3;
