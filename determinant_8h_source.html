39 template<typename T, size_t _rows, size_t _col>
40 auto determinant4x4(const Matrix<T, _rows, _col> &matrix) -> enableIfFloat<T, T>
41 {
42 
43 #if defined(TL_HAVE_SIMD_INTRINSICS) && defined(TL_HAVE_SSE2)
44 
45  __m128 row0 = _mm_load_ps(&matrix(0, 0));
46  __m128 row1 = _mm_load_ps(&matrix(1, 0));
47  __m128 row2 = _mm_load_ps(&matrix(2, 0));
48  __m128 row3 = _mm_load_ps(&matrix(3, 0));
49 
50  // Calcular cofactors de la primera fila
51  __m128 fac0, fac1, fac2, fac3;
52  {
53  __m128 tmp0 = _mm_shuffle_ps(row2, row3, _MM_SHUFFLE(0, 0, 0, 0));
54  __m128 tmp1 = _mm_shuffle_ps(row2, row3, _MM_SHUFFLE(1, 1, 1, 1));
55  __m128 tmp2 = _mm_shuffle_ps(row2, row3, _MM_SHUFFLE(2, 2, 2, 2));
56  __m128 tmp3 = _mm_shuffle_ps(row2, row3, _MM_SHUFFLE(3, 3, 3, 3));
57 
58  __m128 tmp4 = _mm_shuffle_ps(row1, row1, _MM_SHUFFLE(3, 2, 1, 0));
59  __m128 tmp5 = _mm_shuffle_ps(row1, row1, _MM_SHUFFLE(3, 2, 0, 1));
60 
61  fac0 = _mm_sub_ps(_mm_mul_ps(tmp4, tmp3), _mm_mul_ps(tmp5, tmp2));
62  fac1 = _mm_sub_ps(_mm_mul_ps(tmp5, tmp1), _mm_mul_ps(tmp4, tmp0));
63  fac2 = _mm_sub_ps(_mm_mul_ps(tmp4, tmp2), _mm_mul_ps(tmp5, tmp3));
64  fac3 = _mm_sub_ps(_mm_mul_ps(tmp5, tmp0), _mm_mul_ps(tmp4, tmp1));
65  }
66 
67  // Determinante
68  __m128 mul0 = _mm_mul_ps(row0, fac0);
69  __m128 mul1 = _mm_mul_ps(_mm_shuffle_ps(row0, row0, _MM_SHUFFLE(3, 2, 1, 0)), fac1);
70  __m128 mul2 = _mm_mul_ps(_mm_shuffle_ps(row0, row0, _MM_SHUFFLE(3, 0, 2, 1)), fac2);
71  __m128 mul3 = _mm_mul_ps(_mm_shuffle_ps(row0, row0, _MM_SHUFFLE(3, 1, 0, 2)), fac3);
72 
73  __m128 add0 = _mm_add_ps(mul0, _mm_sub_ps(mul1, mul2));
74  __m128 add1 = _mm_add_ps(add0, mul3);
75 
76  float result[4];
77  _mm_store_ps(result, add1);
78 
79  return result[0] - result[1] + result[2] - result[3];
80 
81 #else
82 
83  __m256i row0 = _mm256_loadu_si256((__m256i*)&matrix(0, 0));
84  __m256i row1 = _mm256_loadu_si256((__m256i*)&matrix(1, 0));
85  __m256i row2 = _mm256_loadu_si256((__m256i*)&matrix(2, 0));
86  __m256i row3 = _mm256_loadu_si256((__m256i*)&matrix(3, 0));
87 
88  __m256i tmp1, tmp2, tmp3, tmp4, minor0, minor1, minor2, minor3;
89 
90  // Calcular los cofactors de la primera fila
91  tmp1 = _mm256_shuffle_epi32(row2, _MM_SHUFFLE(0, 0, 3, 3));
92  tmp2 = _mm256_shuffle_epi32(row3, _MM_SHUFFLE(1, 1, 2, 2));
93  tmp3 = _mm256_shuffle_epi32(row2, _MM_SHUFFLE(1, 1, 2, 2));
94  tmp4 = _mm256_shuffle_epi32(row3, _MM_SHUFFLE(0, 0, 3, 3));
95 
96  minor0 = _mm256_sub_epi32(_mm256_mullo_epi32(_mm256_shuffle_epi32(row1, _MM_SHUFFLE(0, 0, 0, 0)), tmp1),
97  _mm256_mullo_epi32(_mm256_shuffle_epi32(row1, _MM_SHUFFLE(3, 3, 3, 3)), tmp2));
98  minor1 = _mm256_sub_epi32(_mm256_mullo_epi32(_mm256_shuffle_epi32(row1, _MM_SHUFFLE(1, 1, 1, 1)), tmp3),
99  _mm256_mullo_epi32(_mm256_shuffle_epi32(row1, _MM_SHUFFLE(2, 2, 2, 2)), tmp4));
100 
101  tmp1 = _mm256_shuffle_epi32(row2, _MM_SHUFFLE(2, 2, 1, 1));
102  tmp2 = _mm256_shuffle_epi32(row3, _MM_SHUFFLE(3, 3, 0, 0));
103  tmp3 = _mm256_shuffle_epi32(row2, _MM_SHUFFLE(3, 3, 0, 0));
104  tmp4 = _mm256_shuffle_epi32(row3, _MM_SHUFFLE(2, 2, 1, 1));
105 
106  minor2 = _mm256_sub_epi32(_mm256_mullo_epi32(_mm256_shuffle_epi32(row1, _MM_SHUFFLE(2, 2, 2, 2)), tmp1),
107  _mm256_mullo_epi32(_mm256_shuffle_epi32(row1, _MM_SHUFFLE(1, 1, 1, 1)), tmp2));
108  minor3 = _mm256_sub_epi32(_mm256_mullo_epi32(_mm256_shuffle_epi32(row1, _MM_SHUFFLE(3, 3, 3, 3)), tmp3),
109  _mm256_mullo_epi32(_mm256_shuffle_epi32(row1, _MM_SHUFFLE(0, 0, 0, 0)), tmp4));
110 
111  // Determinante
112  __m256i mul0 = _mm256_mullo_epi32(row0, minor0);
113  __m256i mul1 = _mm256_mullo_epi32(_mm256_shuffle_epi32(row0, _MM_SHUFFLE(2, 2, 1, 1)), minor1);
114  __m256i mul2 = _mm256_mullo_epi32(_mm256_shuffle_epi32(row0, _MM_SHUFFLE(1, 1, 0, 0)), minor2);
115  __m256i mul3 = _mm256_mullo_epi32(_mm256_shuffle_epi32(row0, _MM_SHUFFLE(3, 3, 2, 2)), minor3);
116 
117  __m256i add0 = _mm256_sub_epi32(mul0, mul1);
118  __m256i add1 = _mm256_add_epi32(add0, mul2);
119  __m256i add2 = _mm256_sub_epi32(add1, mul3);
120 
121  int result[8];
122  _mm256_storeu_si256((__m256i*)result, add2);
123 
124  return result[0] - result[1] + result[2] - result[3];
125 
126 #endif // TL_HAVE_SIMD_INTRINSICS

133 template<typename T, size_t _rows, size_t _col>
134 auto determinant4x4(const Matrix<T, _rows, _col> &matrix) -> enableIfDouble<T, T>
135 {
136 
137 #if defined(TL_HAVE_SIMD_INTRINSICS) && defined(TL_HAVE_AVX)
138 
139  __m256d row0 = _mm256_load_pd(matrix[0][0]);
140  __m256d row1 = _mm256_load_pd(matrix[1][0]);
141  __m256d row2 = _mm256_load_pd(matrix[2][0]);
142  __m256d row3 = _mm256_load_pd(matrix[3][0]);
143 
144  // Calcular cofactors de la primera fila
145  __m256d fac0, fac1, fac2, fac3;
146  {
147  __m256d tmp0 = _mm256_shuffle_pd(row2, row3, 0x0);
148  __m256d tmp1 = _mm256_shuffle_pd(row2, row3, 0x5);
149  __m256d tmp2 = _mm256_shuffle_pd(row2, row3, 0xA);
150  __m256d tmp3 = _mm256_shuffle_pd(row2, row3, 0xF);
151  fac0 = _mm256_sub_pd(_mm256_mul_pd(_mm256_shuffle_pd(row1, row1, 0x0), tmp3),
152  _mm256_mul_pd(_mm256_shuffle_pd(row1, row1, 0x5), tmp2));
153  fac1 = _mm256_sub_pd(_mm256_mul_pd(_mm256_shuffle_pd(row1, row1, 0x5), tmp1),
154  _mm256_mul_pd(_mm256_shuffle_pd(row1, row1, 0x0), tmp0));
155  fac2 = _mm256_sub_pd(_mm256_mul_pd(_mm256_shuffle_pd(row1, row1, 0x0), tmp2),
156  _mm256_mul_pd(_mm256_shuffle_pd(row1, row1, 0x5), tmp3));
157  fac3 = _mm256_sub_pd(_mm256_mul_pd(_mm256_shuffle_pd(row1, row1, 0x5), tmp0),
158  _mm256_mul_pd(_mm256_shuffle_pd(row1, row1, 0x0), tmp1));
159  }
160 
161  // Determinante
162  __m256d mul0 = _mm256_mul_pd(row0, fac0);
163  __m256d mul1 = _mm256_mul_pd(_mm256_shuffle_pd(row0, row0, 0x5), fac1);
164  __m256d mul2 = _mm256_mul_pd(_mm256_shuffle_pd(row0, row0, 0xA), fac2);
165  __m256d mul3 = _mm256_mul_pd(_mm256_shuffle_pd(row0, row0, 0xF), fac3);
166 
167  __m256d add0 = _mm256_add_pd(mul0, _mm256_sub_pd(mul1, mul2));
168  __m256d add1 = _mm256_add_pd(add0, mul3);
169 
170  double result[4];
171  _mm256_store_pd(result, add1);
172 
173  return result[0] - result[1] + result[2] - result[3];
174 
175 #elif defined(TL_HAVE_SIMD_INTRINSICS) && defined(TL_HAVE_SSE2)
176 
177  __m128d row0_lo = _mm_loadu_pd(&matrix[0][0]);
178  __m128d row0_hi = _mm_loadu_pd(&matrix[0][2]);
179  __m128d row1_lo = _mm_loadu_pd(&matrix[1][0]);
180  __m128d row1_hi = _mm_loadu_pd(&matrix[1][2]);
181  __m128d row2_lo = _mm_loadu_pd(&matrix[2][0]);
182  __m128d row2_hi = _mm_loadu_pd(&matrix[2][2]);
183  __m128d row3_lo = _mm_loadu_pd(&matrix[3][0]);
184  __m128d row3_hi = _mm_loadu_pd(&matrix[3][2]);
185 
186  __m128d minor0, minor1, minor2, minor3;
187  __m128d tmp1, tmp2, tmp3, tmp4;
188 
189  // Shuffle rows to align for subdeterminants
190  tmp1 = _mm_shuffle_pd(row2_lo, row3_lo, _MM_SHUFFLE2(0, 0));
191  tmp2 = _mm_shuffle_pd(row2_hi, row3_hi, _MM_SHUFFLE2(0, 0));
192  tmp3 = _mm_shuffle_pd(row2_lo, row3_lo, _MM_SHUFFLE2(1, 1));
193  tmp4 = _mm_shuffle_pd(row2_hi, row3_hi, _MM_SHUFFLE2(1, 1));
194 
195  // Compute cofactors of the first row
196  minor0 = _mm_sub_pd(_mm_mul_pd(_mm_shuffle_pd(row1_lo, row1_lo, _MM_SHUFFLE2(1, 1)), tmp4),
197  _mm_mul_pd(_mm_shuffle_pd(row1_lo, row1_lo, _MM_SHUFFLE2(0, 0)), tmp2));
198  minor1 = _mm_sub_pd(_mm_mul_pd(_mm_shuffle_pd(row1_hi, row1_hi, _MM_SHUFFLE2(1, 1)), tmp3),
199  _mm_mul_pd(_mm_shuffle_pd(row1_hi, row1_hi, _MM_SHUFFLE2(0, 0)), tmp1));
200  minor2 = _mm_sub_pd(_mm_mul_pd(_mm_shuffle_pd(row1_lo, row1_lo, _MM_SHUFFLE2(1, 1)), tmp2),
201  _mm_mul_pd(_mm_shuffle_pd(row1_lo, row1_lo, _MM_SHUFFLE2(0, 0)), tmp4));
202  minor3 = _mm_sub_pd(_mm_mul_pd(_mm_shuffle_pd(row1_hi, row1_hi, _MM_SHUFFLE2(1, 1)), tmp1),
203  _mm_mul_pd(_mm_shuffle_pd(row1_hi, row1_hi, _MM_SHUFFLE2(0, 0)), tmp3));
204 
205  // Combine results
206  __m128d det0 = _mm_add_pd(_mm_shuffle_pd(minor0, minor1, _MM_SHUFFLE2(0, 0)),
207  _mm_shuffle_pd(minor0, minor1, _MM_SHUFFLE2(1, 1)));
208  __m128d det1 = _mm_add_pd(_mm_shuffle_pd(minor2, minor3, _MM_SHUFFLE2(0, 0)),
209  _mm_shuffle_pd(minor2, minor3, _MM_SHUFFLE2(1, 1)));
210 
211  // Multiply first row by cofactors
212  __m128d row0 = _mm_add_pd(row0_lo, row0_hi);
213  __m128d det = _mm_mul_pd(row0, _mm_add_pd(det0, det1));
214 
215  // Sum the components
216  double result[2];
217  _mm_storeu_pd(result, det);
218 
219  return result[0] - result[1];
220 
221 #else
222 
224 
225 #endif
