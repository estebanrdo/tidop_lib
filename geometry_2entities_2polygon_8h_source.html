372 template<typename Point_t>
373 auto Polygon<Point_t>::isInner(const Point_t &point) const -> bool
374 {
375  Window<Point_t> w = this->window();
376  // Comprueba si esta dentro de la ventana envolvente.
377  if (w.containsPoint(point) == false) return false;
378 
379  // Se comprueba si el punto es uno de los vertices
380  for (int i = 0; i < this->size(); i++) {
381  // Por ahora se devuelve true. Lo suyo serÃ­a indicar que es un vertice.
382  if (this->at(i) == point) return true;
383  }
384 
385  Segment<Point_t> sPointH(point, Point_t(w.pt2.x, point.y));
386 
387  // Comprueba si corta una recta que pase por el punto corta en un numero
388  // par al poligono. La recta va desde el punto a la parte inferior, superior,
389  // derecha o izquierda de la ventana.
390  //... Se suma uno porque Mat comienza en cero. Se podria hacer una conversion a cv::Rect
391  //cv::Mat aux(w.height()+1, w.width()+1, CV_8UC3);
392  //cv::line(aux, point-w.pt1, point-w.pt1, Color(Color::NAME::Red).get<cv::Scalar>(), 1);
393  int nIntersection = 0;
394  bool bVertex = false;
395  std::vector<int> vertex_id;
396 
397  for (size_t i = 0, j = 1; i < this->size(); i++, j++) {
398  if (j == this->size()) j = 0;
399 
400 
401  Segment<Point_t> segment(this->at(i), this->at(j));
402 
403  // El punto es colineal con el segmento y esta dentro del mismo.
404  if (distPointToSegment(point, segment) == 0) return true;
405 
406  Point_t ptp;
407  //if (projectPointInSegment(segment, point, &ptp) == 2) return true; // esta en la linea
408 
409 
410  if (point.y == segment.pt1.y) {
411  vertex_id.push_back(static_cast<int>(i));
412  bVertex = true;
413  }
414 
415  if (point.y == segment.pt1.y && point.y == segment.pt2.y) {
416  // los dos segmentos forman parte de la misma linea horizontal
417  // como no va a aparecer intersecciÃ³n...
418  //nIntersection += 2;
419  } else {
420  nIntersection += intersectSegments(segment, sPointH, &ptp);
421  }
422 
423  }
424 
425  // hay vertices
426  // Si hay vertices hay que estudiar mas en detalle si el punto esta dentro o fuera
427  if (bVertex) {
428  std::vector<int> order;
429  int vertex_prev = 0;
430  int vertex_next = 0;
431  for (size_t i = 0; i < vertex_id.size(); i++) {
432  // Se comprueban los puntos anterior y siguiente
433  if (vertex_id[i] == 0) vertex_prev = static_cast<int>(this->size()) - 1;
434  else vertex_prev = vertex_id[i] - 1;
435 
436  if (vertex_id[i] == this->size() - 1) vertex_next = 0;
437  else vertex_next = vertex_id[i] + 1;
438 
439  if (this->at(vertex_prev).y == sPointH.pt1.y) {
440  continue;
441  } else {
442  if (this->at(vertex_next).y == sPointH.pt1.y) {
443  int prev = isLeft(sPointH.pt1, sPointH.pt2, this->at(vertex_prev));
444  if (vertex_next == this->size() - 1) vertex_next = 0;
445  else vertex_next = vertex_next + 1;
446  int next = isLeft(sPointH.pt1, sPointH.pt2, this->at(vertex_next));
447  if (prev == next) nIntersection -= 2;
448  else nIntersection--;
449  } else {
450  int prev = isLeft(sPointH.pt1, sPointH.pt2, this->at(vertex_prev));
451  int next = isLeft(sPointH.pt1, sPointH.pt2, this->at(vertex_next));
452  if (prev == next) nIntersection -= 2;
453  else nIntersection--;
454  }
455 
456  }
457 
458  }
459  }
460  if (nIntersection < 0) return false;
461  if (nIntersection % 2 == 0) return false;
462 
463  return true;
464 
466  //std::vector<cv::Point2f> points;
467  //for (int i = 0; i < size(); i++) {
468  // points.push_back(cv::Point2f(at(i).x, at(i).y));
469  //}
470  //double i_pol_test = cv::pointPolygonTest(mEntities, cv::Point2f(point.x, point.y), false);
471  //if (i_pol_test == -1) return false;
472  //else return true;
473 
474 }

528 template<typename Point_t>
529 auto Polygon<Point_t>::window() const -> Window<Point_t>
530 {
531  Window<Point_t> w;
532 
533  for (size_t i = 0; i < this->size(); i++) {
534  if (w.pt1.x > this->at(i).x) w.pt1.x = this->at(i).x;
535  if (w.pt1.y > this->at(i).y) w.pt1.y = this->at(i).y;
536  if (w.pt2.x < this->at(i).x) w.pt2.x = this->at(i).x;
537  if (w.pt2.y < this->at(i).y) w.pt2.y = this->at(i).y;
538  }
539 
540  return w;
541 }
