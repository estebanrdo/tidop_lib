38 void rgbToCmyk(const cv::Mat &rgb, cv::Mat &cmyk)
39 {
40  try {
41 
42  TL_ASSERT(rgb.channels() == 3, "Invalid image type. Incorrect number of channels");
43 
44  cv::Mat _cmyk(rgb.size(), CV_32FC4);
45 
46  parallel_for(static_cast<size_t>(0), static_cast<size_t>(rgb.rows), [&](size_t row) {
47 
48  double cyan = 0.;
49  double magenta = 0.;
50  double yellow = 0.;
51  double key = 1.;
52 
53  int r = static_cast<int>(row);
54  const uchar *rgb_ptr = rgb.ptr<uchar>(r);
55 
56  for (int c = 0; c < rgb.cols; c++) {
57 
58  rgbToCmyk(rgb_ptr[3 * static_cast<size_t>(c) + 2],
59  rgb_ptr[3 * static_cast<size_t>(c) + 1],
60  rgb_ptr[3 * static_cast<size_t>(c)],
61  &cyan,
62  &magenta,
63  &yellow,
64  &key);
65 
66  _cmyk.at<cv::Vec4f>(r, c) = cv::Vec4f(static_cast<float>(cyan),
67  static_cast<float>(magenta),
68  static_cast<float>(yellow),
69  static_cast<float>(key));
70  }
71  });
72 
73  cmyk = _cmyk;
74 
75  } catch (...) {
76  std::throw_with_nested(std::runtime_error("rgbToCmyk() failed"));
77  }

350 void ColorConversion::run(const cv::Mat &matIn, cv::Mat &matOut) const
351 {
352  try {
353 
354  TL_ASSERT(!matIn.empty(), "Incorrect input data. Empty image");
355 
356  if (mModelIn == ColorModel::rgb &&
357  mModelOut == ColorModel::hsl) {
358  rgbToHSL(matIn, matOut);
359  } else if (mModelIn == ColorModel::hsl &&
360  mModelOut == ColorModel::rgb) {
361  hslToRgb(matIn, matOut);
362  } else if (mModelIn == ColorModel::rgb &&
363  mModelOut == ColorModel::hsv) {
364  rgbToHSV(matIn, matOut);
365  } else if (mModelIn == ColorModel::hsv &&
366  mModelOut == ColorModel::rgb) {
367  hsvToRgb(matIn, matOut);
368  } else if (mModelIn == ColorModel::rgb &&
369  mModelOut == ColorModel::cmyk) {
370  rgbToCmyk(matIn, matOut);
371  } else if (mModelIn == ColorModel::cmyk &&
372  mModelOut == ColorModel::rgb) {
373  cmykToRgb(matIn, matOut);
374  } else if (mModelIn == ColorModel::rgb &&
375  mModelOut == ColorModel::luminance) {
376  rgbToLuminance(matIn, matOut);
377  } else {
378  throw TL_ERROR("Color conversion not available");
379  }
380 
381  } catch (...) {
382  std::throw_with_nested(std::runtime_error("ColorConversion::run() failed"));
