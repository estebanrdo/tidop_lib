140 auto SingularValueDecomposition<Matrix_t<T, Rows, Cols>>::solve(const Vector<T, Rows>& B) -> Vector<T, Cols>
141 {
142  Vector<T, Cols> C(mCols);
143 
144  T s;
145  Vector<T, Cols> tmp(mCols);
146 
147  for (size_t j = 0; j < mCols; j++) {
148  s = consts::zero<T>;
149  if (W[j] > tsh) {
150  for (size_t i = 0; i < mRows; i++)
151  s += U[i][j] * B[i];
152  s /= W[j];
153  }
154  tmp[j] = s;
155  }
156 
157  for (size_t j = 0; j < mCols; j++) {
158  s = consts::zero<T>;
159  for (size_t k = 0; k < mCols; k++)
160  s += V[j][k] * tmp[k];
161  C[j] = s;
162  }
163 
164  return C;
165 }

167 template<
168  template<typename, size_t, size_t>
169 class Matrix_t, typename T, size_t Rows, size_t Cols
170 >
171 void SingularValueDecomposition<Matrix_t<T, Rows, Cols>>::decompose()
172 {
173  this->U = this->A;
174  Vector<T, Cols> rv1(mCols);
175  int i;
176  int j;
177  int k;
178  int l;
179  T anorm;
180  T f;
181  T g;
182  T h;
183  T s;
184  T scale;
185  g = scale = anorm = consts::zero<T>; //Householder reduction to bidiagonal form.
186 
187  //int one = 1;
188 
189  for (i = 0; i < static_cast<int>(mCols); i++) {
190  l = i + 2;
191  rv1[i] = scale * g;
192  g = s = scale = 0.0;
193 
194  if (i < static_cast<int>(mRows)) {
195  for (k = i; k < static_cast<int>(mRows); k++) scale += std::abs(U[k][i]);
196  if (scale != 0.) {
197  for (k = i; k < static_cast<int>(mRows); k++) {
198  U[k][i] /= scale;
199  s += U[k][i] * U[k][i];
200  }
201  f = U[i][i];
202  g = -std::copysign(sqrt(s), f);
203  h = f * g - s;
204  U[i][i] = f - g;
205  for (j = l - consts::one<int>; j < static_cast<int>(mCols); j++) {
206  for (s = 0.0, k = i; k < static_cast<int>(mRows); k++)
207  s += U[k][i] * U[k][j];
208  f = s / h;
209  for (k = i; k < mRows; k++)
210  U[k][j] += f * U[k][i];
211  }
212  for (k = i; k < mRows; k++)
213  U[k][i] *= scale;
214  }
215  }
216 
217  W[i] = scale * g;
218  g = s = scale = 0.0;
219  if (i + consts::one<int> <= mRows && i + consts::one<int> != mCols) {
220  for (k = l - consts::one<int>; k < mCols; k++) scale += fabs(U[i][k]);
221  if (scale != 0.) {
222  for (k = l - consts::one<int>; k < mCols; k++) {
223  U[i][k] /= scale;
224  s += U[i][k] * U[i][k];
225  }
226  f = U[i][l - consts::one<int>];
227  g = -std::copysign(sqrt(s), f);
228  h = f * g - s;
229  U[i][l - consts::one<int>] = f - g;
230 
231  for (k = l - consts::one<int>; k < mCols; k++)
232  rv1[k] = U[i][k] / h;
233 
234  for (j = l - consts::one<int>; j < mRows; j++) {
235  for (s = 0.0, k = l - consts::one<int>; k < mCols; k++)
236  s += U[j][k] * U[i][k];
237  for (k = l - consts::one<int>; k < mCols; k++)
238  U[j][k] += s * rv1[k];
239  }
240 
241  for (k = l - consts::one<int>; k < mCols; k++)
242  U[i][k] *= scale;
243  }
244  }
245  anorm = std::max(anorm, std::abs(W[i]) + std::abs(rv1[i]));
246  }
247 
248  for (i = static_cast<int>(mCols) - consts::one<int>; i >= 0; i--) { //Accumulation of right-hand transformations.
249  if (i < mCols - consts::one<int>) {
250  if (g != 0.) {
251  for (j = l; j < mCols; j++) //Double division to avoid possible underflow.
252  V[j][i] = (U[i][j] / U[i][l]) / g;
253  for (j = l; j < mCols; j++) {
254  for (s = 0.0, k = l; k < mCols; k++)
255  s += U[i][k] * V[k][j];
256  for (k = l; k < mCols; k++)
257  V[k][j] += s * V[k][i];
258  }
259  }
260 
261  for (j = l; j < mCols; j++)
262  V[i][j] = V[j][i] = consts::zero<T>;
263  }
264 
265  V[i][i] = consts::one<T>;
266  g = rv1[i];
267  l = i;
268  }
269 
270  for (i = static_cast<int>(std::min(mRows, mCols)) - consts::one<int>; i >= 0; i--) { //Accumulation of left-hand transformations.
271  l = i + consts::one<int>;
272  g = W[i];
273  for (j = l; j < mCols; j++) this->U.at(i, j) = 0.0;
274  if (g != consts::zero<T>) {
275  g = consts::one<T> / g;
276  for (j = l; j < mCols; j++) {
277  for (s = consts::zero<T>, k = l; k < mRows; k++)
278  s += U[k][i] * U[k][j];
279  f = (s / U[i][i]) * g;
280  for (k = i; k < mRows; k++)
281  U[k][j] += f * U[k][i];
282  }
283  for (j = i; j < mRows; j++)
284  U[j][i] *= g;
285  } else {
286  for (j = i; j < mRows; j++)
287  U[j][i] = consts::zero<T>;
288  }
289  ++U[i][i];
290  }
291 
292  bool flag;
293  int its;
294  size_t nm;
295  T c;
296  T x;
297  T y;
298  T z;
299  for (k = static_cast<int>(mCols) - consts::one<int>; k >= 0; k--) { //Diagonalization of the bidiagonal form: Loop over
300  for (its = 0; its < mIterationMax; its++) { // singular values, and over allowed iterations.
301 
302  flag = true;
303 
304  for (l = k; l >= 0; l--) { //Test for splitting.
305  nm = static_cast<size_t>(l) - consts::one<int>; //Note that rv1[1] is always zero.
306  if (l == 0 || std::abs(rv1[l]) <= eps * anorm) {
307  flag = false;
308  break;
309  }
310  if (std::abs(W[nm]) <= eps * anorm) break;
311  }
312 
313  if (flag) {
314  c = consts::zero<T>; //Cancellation of rv1[l], if l > 1.
315  s = consts::one<T>;
316  for (i = l; i < k + consts::one<int>; i++) {
317  f = s * rv1[i];
318  rv1[i] = c * rv1[i];
319  if (std::abs(f) <= eps * anorm) break;
320  g = W[i];
321  h = module(f, g);
322  W[i] = h;
323  h = consts::one<T> / h;
324  c = g * h;
325  s = -f * h;
326  for (j = 0; j < mRows; j++) {
327  y = U[j][nm];
328  z = U[j][i];
329  U[j][nm] = y * c + z * s;
330  U[j][i] = z * c - y * s;
331  }
332  }
333  }
334  z = W[k];
335  if (l == k) { //Convergence.
336  if (z < consts::zero<T>) { //Singular value is made nonnegative.
337  W[k] = -z;
338  for (j = 0; j < mCols; j++)
339  V[j][k] = -V[j][k];
340  }
341  break;
342  }
343 
344  if (its == 29) throw std::runtime_error("no convergence in 30 iterations");
345  x = W[l]; //Shift from bottom 2-by-2 minor.
346  nm = static_cast<size_t>(k - consts::one<int>);
347  y = W[nm];
348  g = rv1[nm];
349  h = rv1[k];
350  f = ((y - z) * (y + z) + (g - h) * (g + h)) / (static_cast<T>(2) * h * y);
351  g = module(f, consts::one<T>);
352  f = ((x - z) * (x + z) + h * ((y / (f + std::copysign(g, f))) - h)) / x;
353  c = s = consts::one<T>; //Next QR transformation:
354 
355  for (j = l; j <= nm; j++) {
356  i = j + consts::one<int>;
357  g = rv1[i];
358  y = W[i];
359  h = s * g;
360  g = c * g;
361  z = module(f, h);
362  rv1[j] = z;
363  c = f / z;
364  s = h / z;
365  f = x * c + g * s;
366  g = g * c - x * s;
367  h = y * s;
368  y *= c;
369  for (size_t jj = 0; jj < mCols; jj++) {
370  x = V[jj][j];
371  z = V[jj][i];
372  V[jj][j] = x * c + z * s;
373  V[jj][i] = z * c - x * s;
374  }
375  z = module(f, h);
376  W[j] = z; //Rotation can be arbitrary if z = 0.
377  if (z != 0.) {
378  z = consts::one<T> / z;
379  c = f * z;
380  s = h * z;
381  }
382  f = c * g + s * y;
383  x = c * y - s * g;
384  for (size_t jj = 0; jj < mRows; jj++) {
385  y = U[jj][j];
386  z = U[jj][i];
387  U[jj][j] = y * c + z * s;
388  U[jj][i] = z * c - y * s;
389  }
390  }
391 
392  rv1[l] = consts::zero<T>;
393  rv1[k] = f;
394  W[k] = x;
395  }
396  }
397 }

401 class Matrix_t, typename T, size_t Rows, size_t Cols
402 >
403 void SingularValueDecomposition<Matrix_t<T, Rows, Cols>>::reorder()
404 {
405  int i;
406  int j;
407  size_t k;
408 
409  int inc = 1;
410  T sw;
411  Vector<T, Rows> su(mRows);
412  Vector<T, Cols> sv(mCols);
413 
414  do { inc *= 3; inc++; } while (inc <= mCols);
415 
416  do {
417  inc /= 3;
418  for (i = inc; i < mCols; i++) {
419  sw = W[i];
420  for (k = 0; k < mRows; k++) su[k] = U.at(k, i);
421  for (k = 0; k < mCols; k++) sv[k] = V.at(k, i);
422  j = i;
423  while (W[j - inc] < sw) {
424  W[j] = W[j - inc];
425  for (k = 0; k < mRows; k++) U.at(k, j) = U.at(k, j - inc);
426  for (k = 0; k < mCols; k++) V.at(k, j) = V.at(k, j - inc);
427  j -= inc;
428  if (j < inc) break;
429  }
430  W[j] = sw;
431  for (k = 0; k < mRows; k++) U.at(k, j) = su[k];
432  for (k = 0; k < mCols; k++) V.at(k, j) = sv[k];
433 
434  }
435  } while (inc > 1);
436 
437  size_t s;
438  for (size_t c = 0; c < mCols; c++) {
439  s = 0;
440 
441  for (size_t r = 0; r < mRows; r++)
442  if (U[r][c] < consts::zero<T>)
443  s++;
444 
445  for (size_t r = 0; r < mCols; r++)
446  if (V[r][c] < consts::zero<T>)
447  s++;
448 
449  if (s > (mRows + mCols) / 2) {
450  U.col(c) = -U.col(c);
451  V.col(c) = -V.col(c);
452  }
453  }
454 }
