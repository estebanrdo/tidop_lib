168 auto ImageMetadataBase::metadata(const std::string& name, bool& active) const -> std::string
169 {
170  std::string value;
171  active = false;
172 
173  auto metadata = mMetadata.find(name);
174  if (metadata != mMetadata.end()) {
175  value = metadata->second.first;
176  active = metadata->second.second;
177  } else {
178  Message::warning("Metadata '{}' not supported", name);
179  }
180 
181  return value;
182 }
183 
184 void ImageMetadataBase::setMetadata(const std::string &name, const std::string &value)
185 {
186  auto metadata = mMetadata.find(name);
187  if (metadata != mMetadata.end()) {
188  metadata->second.first = value;
189  metadata->second.second = true;
190  } /*else {
191  msgWarning("Metadata '%s' not supported", name.c_str());
192  }*/
193 }
194 
195 auto ImageMetadataBase::metadata(bool all) const -> std::map<std::string, std::string>
196 {
197  std::map<std::string, std::string> metadata;
198 
199  for (auto &pair : mMetadata) {
200  if (all || pair.second.second == true)
201  metadata[pair.first] = pair.second.first;
202  }
203 
204  return metadata;
205 }
